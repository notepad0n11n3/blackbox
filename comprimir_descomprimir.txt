  ::: 7z l archivo.zip         // (si... l sin - )

ARCHIVOS .zip
  Comprimir:
    ::: zip archivo.zip carpeta
  Descomprimir:
    ::: unzip archivo.zip
    ::: unzip archivo.zip -d carpetaName

ARCHIVOS .tar.gz
  Comprimir:
    ::: tar -czvf empaqutado.tr.gz  /carpeta/a/empaquetar/
  Descomprimier:
    ::: tar -xzvf archivo.tar.gz

ARCHIVOS .tar
  Empaquetar:
    ::: tar -cvf paquete.tar  /dir/a/comprimir/
  Desempaquetar:
    ::: tar -xvf paquete.tar

ARCHIVOS .gz
  Comprimir:
    ::: gzip -9 index.php
  Descomprimir:
    ::: gzip -d index.php.gz
